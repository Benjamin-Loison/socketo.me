 11:  * Send {@see RequestInterface} objects in parallel using curl_multi
 12:  *
 13:  * This implementation allows callers to send blocking requests that return back
 14:  * to the caller when their requests complete, regardless of whether or not
 15:  * previously sending requests in the curl_multi object have completed.  The

 17:  * a request to the CurlMulti object, and tracking the requests in the current

147:      * {@inheritdoc}
148:      *
149:      * Adds a request to a batch of requests to be sent in parallel.
150:      *
151:      * Async requests adds a request to the current scope to be executed in
152:      * parallel with any currently executing cURL handles.  You may only add an
153:      * async request while other requests are transferring.  Attempting to add
154:      * an async request while no requests are transferring will add the request
155:      * normally in the next available scope (typically 0).

260:      * {@inheritdoc}
261:      */
262:     public function send()
263:     {
264:         $this->scope++;
265:         $this->state = self::STATE_SENDING;
266: 
267:         // Only prepare and send requests that are in the current recursion scope
268:         // Only enter the main perform() loop if there are requests in scope
269:         if (!empty($this->requests[$this->scope])) {
270: 
271:             // Any exceptions thrown from this event should break the entire
272:             // flow of sending requests in parallel to prevent weird errors
273:             $this->dispatch(self::BEFORE_SEND, array(
274:                 'requests' => $this->requests[$this->scope]
275:             ));
276: 
277:             foreach ($this->requests[$this->scope] as $request) {
278:                 if ($request->getState() != RequestInterface::STATE_TRANSFER) {
279:                     $this->beforeSend($request);
280:                 }
281:             }
282: 
283:             try {
284:                 $this->perform();
285:             } catch (\Exception $e) {
286:                 $this->exceptions[] = $e;
287:             }
288:         }
289: 
290:         $this->scope--;
291: 
292:         // Aggregate exceptions into an ExceptionCollection
293:         $exceptionCollection = null;
294:         if (!empty($this->exceptions)) {
295:             $exceptionCollection = new ExceptionCollection('Errors during multi transfer');
296:             while ($e = array_shift($this->exceptions)) {
297:                 $exceptionCollection->add($e);
298:             }
299:         }
300: 
301:         // Complete the transfer if this is not a nested scope
302:         if ($this->scope == -1) {
303:             $this->state = self::STATE_COMPLETE;
304:             $this->dispatch(self::COMPLETE);
305:             $this->reset();
306:         }
307: 
308:         // Throw any exceptions that were encountered
309:         if ($exceptionCollection) {
310:             throw $exceptionCollection;
311:         }
312:     }

367:     /**
368:      * Get the data from the multi handle
369:      */
370:     protected function perform()
371:     {
372:         // @codeCoverageIgnoreStart
373:         // Weird things can happen when making HTTP requests in __destruct methods
374:         if (!$this->multiHandle) {
375:             return;
376:         }
377:         // @codeCoverageIgnoreEnd
378: 
379:         // If there are no requests to send, then exit from the function
380:         if ($this->scope <= 0) {
381:             if ($this->count() == 0) {
382:                 return;
383:             }
384:         } elseif (empty($this->requests[$this->scope])) {
385:             return;
386:         }
387: 
388:         // Create the polling event external to the loop
389:         $event = array('curl_multi' => $this);
390: 
391:         while (1) {
392: 
393:             $active = $this->executeHandles();
394: 
395:             // Get messages from curl handles
396:             while ($done = curl_multi_info_read($this->multiHandle)) {
397:                 $request = $this->resourceHash[(int) $done['handle']];
398:                 $handle = $this->handles[$request];
399:                 try {
400:                     $this->processResponse($request, $handle, $done);
401:                 } catch (\Exception $e) {
402:                     $this->removeErroredRequest($request, $e);
403:                 }
404:             }
405: 
406:             // Notify each request as polling and handled queued responses
407:             $scopedPolling = $this->scope <= 0 ? $this->all() : $this->requests[$this->scope];
408: 
409:             // Exit the function if there are no more requests to send
410:             if (empty($scopedPolling)) {
411:                 break;
412:             }
413: 
414:             // Notify all requests that requests are being polled
415:             foreach ($scopedPolling as $request) {
416:                 $event['request'] = $request;
417:                 $request->dispatch(self::POLLING_REQUEST, $event);
418:             }
419: 
420:             if ($active) {
421:                 // Select the curl handles until there is any activity on any of the open file descriptors
422:                 // See https://github.com/php/php-src/blob/master/ext/curl/multi.c#L170
423:                 $active = $this->executeHandles(true, 0.25);
424:             } else {
425:                 // Sleep to prevent eating CPU because no requests are actually pending a select call
426:                 usleep(500);
427:             }
428:         }
429:     }

431:     /**
432:      * Execute and select curl handles until there is activity
433:      *
434:      * @param bool $select  Set to TRUE to select the file descriptors
435:      * @param int  $timeout Select timeout in seconds
436:      *
437:      * @return int Returns the number of active handles
438:      */
439:     private function executeHandles($select = false, $timeout = 1)
440:     {
441:         $active = $selectResult = 0;
442: 
443:         do {
444: 
445:             if ($select) {
446:                 $selectResult = curl_multi_select($this->multiHandle, $timeout);
447:             }
448: 
449:             if ($selectResult === 0) {
450:                 while ($mrc = curl_multi_exec($this->multiHandle, $active) == CURLM_CALL_MULTI_PERFORM);
451:                 // Check the return value to ensure an error did not occur
452:                 $this->checkCurlResult($mrc);
453:             }
454: 
455:         // Poll once if not selecting, or poll until there are no handles with activity
456:         } while ($select && $active && $selectResult == 0);
457: 
458:         return $active;
459:     }

478:     /**
479:      * Check for errors and fix headers of a request based on a curl response
480:      *
481:      * @param RequestInterface $request Request to process
482:      * @param CurlHandle       $handle  Curl handle object
483:      * @param array            $curl    Array returned from curl_multi_info_read
484:      *
485:      * @throws CurlException on Curl error
486:      */
487:     protected function processResponse(RequestInterface $request, CurlHandle $handle, array $curl)
488:     {
489:         // Set the transfer stats on the response
490:         $handle->updateRequestFromTransfer($request);
491:         // Check if a cURL exception occurred, and if so, notify things
492:         $curlException = $this->isCurlException($request, $handle, $curl);
493: 
494:         // Always remove completed curl handles.  They can be added back again
495:         // via events if needed (e.g. ExponentialBackoffPlugin)
496:         $this->removeHandle($request);
497: 
498:         if (!$curlException) {
499:             $request->setState(RequestInterface::STATE_COMPLETE);
500:             // Only remove the request if it wasn't resent as a result of the state change
501:             if ($request->getState() != RequestInterface::STATE_TRANSFER) {
502:                 $this->remove($request);
503:             }
504:         } else {
505:             // Set the state of the request to an error
506:             $request->setState(RequestInterface::STATE_ERROR);
507:             // Notify things that listen to the request of the failure
508:             $request->dispatch('request.exception', array(
509:                 'request'   => $this,
510:                 'exception' => $curlException
511:             ));
512: 
513:             // Allow things to ignore the error if possible
514:             $state = $request->getState();
515:             if ($state != RequestInterface::STATE_TRANSFER) {
516:                 $this->remove($request);
517:             }
518:             // The error was not handled, so fail
519:             if ($state == RequestInterface::STATE_ERROR) {
520:                 throw $curlException;
521:             }
522:         }
523:     }

525:     /**
526:      * Remove a curl handle from the curl multi object
527:      *
528:      * Nasty things (bus errors, segmentation faults) can sometimes occur when removing curl handles when in a callback
529:      * or a recursive scope.  Here we are queueing all curl handles that need to be removed and closed so that this
530:      * happens only in the outermost scope when everything has completed sending.
531:      *
532:      * @param RequestInterface $request Request that owns the handle
533:      */
534:     protected function removeHandle(RequestInterface $request)
535:     {
536:         if ($this->handles->contains($request)) {
537:             $handle = $this->handles[$request];
538:             unset($this->resourceHash[(int) $handle->getHandle()]);
539:             unset($this->handles[$request]);
540:             $this->removeHandles[] = $handle;
541:         }
542:     }

544:     /**
545:      * Check if a cURL transfer resulted in what should be an exception
546:      *
547:      * @param RequestInterface $request Request to check
548:      * @param CurlHandle       $handle  Curl handle object
549:      * @param array            $curl    Array returned from curl_multi_info_read
550:      *
551:      * @return Exception|bool
552:      */
553:     private function isCurlException(RequestInterface $request, CurlHandle $handle, array $curl)
554:     {
555:         if (CURLE_OK == $curl['result']) {
556:             return false;
557:         }
558: 
559:         $handle->setErrorNo($curl['result']);

563:         $e->setCurlHandle($handle)

565:           ->setError($handle->getError(), $handle->getErrorNo());
566: 
567:         return $e;
