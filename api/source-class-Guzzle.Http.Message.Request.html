603:     /**
604:      * Method to receive HTTP response headers as they are retrieved
605:      *
606:      * @param string $data Header data.
607:      *
608:      * @return integer Returns the size of the data.
609:      */
610:     public function receiveResponseHeader($data)
611:     {
612:         $this->state = self::STATE_TRANSFER;
613:         $length = strlen($data);
614: 
615:         // Normalize line endings
616:         $data = preg_replace("/([^\r])(\n)\b/", "$1\r\n", $data);
617: 
618:         if (preg_match('/^\HTTP\/1\.[0|1]\s\d{3}\s.+$/', $data)) {
619:             $previousResponse = $this->response;
620:             list($dummy, $code, $status) = explode(' ', str_replace("\r\n", '', $data), 3);
621:             $this->response = new Response($code, null, $this->getResponseBody());
622:             $this->response->setStatus($code, $status)->setRequest($this);
623:             $this->dispatch('request.receive.status_line', array(
624:                 'line' => $data,
625:                 'status_code' => $code,
626:                 'reason_phrase' => $status,
627:                 'previous_response' => $previousResponse
628:             ));
629:         } else if ($length > 2) {
630:             list($header, $value) = array_map('trim', explode(':', trim($data), 2));
631:             $this->response->addHeaders(array(
632:                 $header => $value
633:             ));
634:             $this->dispatch('request.receive.header', array(
635:                 'header' => $header,
636:                 'value' => $value
637:             ), true);
638:         }
639: 
640:         return $length;
641:     }
