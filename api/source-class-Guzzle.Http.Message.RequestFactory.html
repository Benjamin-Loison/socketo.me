 64:     public function parseMessage($message)
 65:     {
 66:         if (!$message) {
 67:             return false;
 68:         }
 69: 
 70:         $headers = new Collection();
 71:         $scheme = $host = $body = $method = $user = $pass = $query = $port = $version = $protocol = '';
 72:         $path = '/';
 73: 
 74:         // Inspired by https://github.com/kriswallsmith/Buzz/blob/message-interfaces/lib/Buzz/Message/Parser/Parser.php#L16
 75:         $lines = preg_split('/(\\r?\\n)/', $message, -1, PREG_SPLIT_DELIM_CAPTURE);
 76:         for ($i = 0, $c = count($lines); $i < $c; $i += 2) {
 77: 
 78:             $line = $lines[$i];
 79: 
 80:             // If two line breaks were encountered, then this is the body
 81:             if (empty($line)) {
 82:                 $body = implode('', array_slice($lines, $i + 2));
 83:                 break;
 84:             }
 85: 
 86:             // Parse message headers
 87:             if (!$method) {
 88:                 list($method, $path, $proto) = explode(' ', $line);
 89:                 $method = strtoupper($method);
 90:                 list($protocol, $version) = explode('/', strtoupper($proto));
 91:                 $scheme = 'http';
 92:             } else if (strpos($line, ':')) {
 93:                 list($key, $value) = explode(':', $line, 2);
 94:                 $key = trim($key);
 95:                 // Normalize standard HTTP headers
 96:                 if (in_array(strtolower($key), static::$requestHeaders)) {
 97:                     $key = str_replace(' ', '-', ucwords(str_replace('-', ' ', $key)));
 98:                 }
 99:                 // Headers are case insensitive
100:                 $headers->add($key, trim($value));
101:             }
102:         }
103: 
104:         // Check for the Host header
105:         if (isset($headers['Host'])) {
106:             $host = $headers['Host'];
107:         }
108: 
109:         if (strpos($host, ':')) {
110:             list($host, $port) = array_map('trim', explode(':', $host));
111:             if ($port == 443) {
112:                 $scheme = 'https';
113:             }
114:         } else {
115:             $port = '';
116:         }
117: 
118:         // Check for basic authorization
119:         $auth = isset($headers['Authorization']) ? $headers['Authorization'] : '';
120: 
121:         if ($auth) {
122:             list($type, $data) = explode(' ', $auth);
123:             if (strtolower($type) == 'basic') {
124:                 $data = base64_decode($data);
125:                 list($user, $pass) = explode(':', $data);
126:             }
127:         }
128: 
129:         // Check if a query is present
130:         $qpos = strpos($path, '?');
131:         if ($qpos) {
132:             $query = substr($path, $qpos);
133:             $path = substr($path, 0, $qpos);
134:         }
135: 
136:         return array(
137:             'method' => $method,
138:             'protocol' => $protocol,
139:             'protocol_version' => $version,
140:             'parts' => array(
141:                 'scheme' => $scheme,
142:                 'host' => $host,
143:                 'port' => $port,
144:                 'user' => $user,
145:                 'pass' => $pass,
146:                 'path' => $path,
147:                 'query' => $query
148:             ),
149:             'headers' => $headers->getAll(),
150:             'body' => $body
151:         );
152:     }

158:     {
159:         $parsed = $this->parseMessage($message);
160: 
161:         if (!$parsed) {
162:             return false;
163:         }
164:

166:             $parsed['headers'], $parsed['body'], $parsed['protocol'],
167:             $parsed['protocol_version']);
168: 
169:         // EntityEnclosingRequest adds an "Expect: 100-Continue" header when
170:         // using a raw request body for PUT or POST requests. This factory
171:         // method should accurately reflect the message, so here we are
172:         // removing the Expect header if one was not supplied in the message.
173:         if (!isset($parsed['headers']['Expect'])) {
174:             $request->removeHeader('Expect');
175:         }

190:      * {@inheritdoc}

192:     public function create($method, $url, $headers = null, $body = null)
193:     {
194:         if ($method != 'POST' && $method != 'PUT' && $method != 'PATCH') {
195:             $c = $this->requestClass;
196:             $request = new $c($method, $url, $headers);
197:             if ($body) {
198:                 $request->setResponseBody(EntityBody::factory($body));
199:             }
200:         } else {
201:             $c = $this->entityEnclosingRequestClass;
202:             $request = new $c($method, $url, $headers);
203: 
204:             if ($body) {
205:                 if ($method == 'POST' && (is_array($body) || $body instanceof Collection)) {
206:                     $request->addPostFields($body);
207:                 } else if (is_resource($body) || $body instanceof EntityBody) {
208:                     $request->setBody($body, (string) $request->getHeader('Content-Type'));
209:                 } else {
210:                     $request->setBody((string) $body, (string) $request->getHeader('Content-Type'));
211:                 }
212:             }
213: 
214:             // Fix chunked transfers based on the passed headers
215:             if (isset($headers['Transfer-Encoding']) && $headers['Transfer-Encoding'] == 'chunked') {
216:                 $request->removeHeader('Content-Length')
217:                         ->setHeader('Transfer-Encoding', 'chunked');
218:             }
219:         }
220: 
221:         return $request;
