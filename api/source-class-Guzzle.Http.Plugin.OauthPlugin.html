106:      * Parameters sorted and filtered in order to properly sign a request

114:     {
115:         $params = new Collection(array(

118:             'oauth_signature_method' => $this->config['signature_method'],
119:             'oauth_timestamp'        => $timestamp,
120:             'oauth_version'          => $this->config['version']
121:         ));
122:         // Filter out oauth_token during temp token step, as in request_token.
123:         if ($this->config['token'] !== false) {
124:             $params->add('oauth_token', $this->config['token']);
125:         }
126: 
127:         // Add query string parameters
128:         $params->merge($request->getQuery());
129:         // Add POST fields to signing string
130:         if ($request instanceof EntityEnclosingRequestInterface
131:             && $request->getHeader('Content-Type') == 'application/x-www-form-urlencoded') {
132:             $params->merge($request->getPostFields());

146:      * @param int              $timestamp Timestamp to use for nonce

148:      * @return string

150:     public function getStringToSign(RequestInterface $request, $timestamp)
151:     {
152:         $params = $this->getParamsToSign($request, $timestamp);
153: 
154:         // Build signing string from combined params
155:         $parameterString = array();
156:         foreach ($params as $key => $values) {
157:             $key = rawurlencode($key);
158:             $values = (array) $values;
159:             sort($values);
160:             foreach ($values as $value) {
161:                 if (is_bool($value)) {
162:                     $value = $value ? 'true' : 'false';
163:                 }
164:                 $parameterString[] = $key . '=' . rawurlencode($value);
165:             }
166:         }
167: 
168:         $url = Url::factory($request->getUrl())->setQuery('')->setFragment('');
169: 
170:         return strtoupper($request->getMethod()) . '&'
171:              . rawurlencode($url) . '&'
172:              . rawurlencode(implode('&', $parameterString));
173:     }
