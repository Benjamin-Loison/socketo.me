 51:      * supplied name.  If no dynamic command is found, the client will look for
 52:      * a concrete command class exists matching the name supplied.  If neither
 53:      * are found, an InvalidArgumentException is thrown.
 54:      *
 55:      * @param string $name Name of the command to retrieve
 56:      * @param array $args (optional) Arguments to pass to the command

 58:      * @return CommandInterface
 59:      * @throws InvalidArgumentException if no command can be found by name
 60:      */

 62:     {
 63:         $command = null;
 64: 
 65:         // If a service description is present, see if a command is defined
 66:         if ($this->serviceDescription && $this->serviceDescription->hasCommand($name)) {
 67:             $command = $this->serviceDescription->createCommand($name, $args);
 68:         }
 69: 
 70:         // Check if a concrete command exists using inflection
 71:         if (!$command) {
 72:             // Determine the class to instantiate based on the namespace of the
 73:             // current client and the default location of commands
 74:             $prefix = $this->getConfig('command.prefix');
 75:             if (!$prefix) {
 76:                 // The prefix can be specified in a factory method and is cached
 77:                 $prefix = implode('\\', array_slice(explode('\\', get_class($this)), 0, -1)) . '\\Command\\';
 78:                 $this->getConfig()->set('command.prefix', $prefix);
 79:             }

 84:             if (class_exists($class)) {

 91:         }

102:     /**
103:      * Execute a command and return the response
104:      *
105:      * @param CommandInterface|CommandSet|array $command Command or set to execute
106:      *
107:      * @return mixed Returns the result of the executed command's

114:     public function execute($command)
115:     {
116:         if ($command instanceof CommandInterface) {
117:             $command->setClient($this)->prepare();
118:             $this->dispatch('command.before_send', array(

121:             $command->getRequest()->send();

123:                 'command' => $command

125:             return $command->getResult();
126:         } else if ($command instanceof CommandSet) {
127:             foreach ($command as $c) {
128:                 if ($c->getClient() && $c->getClient() !== $this) {

133:                 }

135:             }

139:         }

141:         throw new \InvalidArgumentException('Invalid command sent to ' . __METHOD__);
142:     }
