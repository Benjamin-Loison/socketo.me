 13:  * The Server class is used to control a scripted webserver using node.js that
 14:  * will respond to HTTP requests with queued responses.
 15:  * 
 16:  * Queued responses will be served to requests using a FIFO order.  All requests
 17:  * received by the server are stored on the node.js server and can be retreived
 18:  * by calling {@see Server::getReceivedRequests()}.
 19:  * 
 20:  * Mock responses that don't require data to be transmitted over HTTP a great
 21:  * for testing.  Mock respones, however, cannot test the actual sending of an
 22:  * HTTP request using cURL.  This test server allows the simulation of any
 23:  * number of HTTP request response transactions to test the actual sending of
 24:  * requests over the wire without having to leave an internal network.

 90:      * Queue an array of responses or a single response on the server.
 91:      *
 92:      * Any currently queued responses will be overwritten.  Subsequent requests
 93:      * on the server will return queued responses in FIFO order.
 94:      *
 95:      * @param array|Response $responses A single or array of Responses to queue
 96:      *
 97:      * @return bool Returns TRUE on success or FALSE on failure
 98:      * @throws HttpException
 99:      */
100:     public function enqueue($responses)
101:     {
102:         $data = array();
103:         foreach ((array) $responses as $response) {
104: 
105:             // Create the response object from a string
106:             if (is_string($response)) {
107:                 $response = Response::factory($response);
108:             } else if (!($response instanceof Response)) {
109:                 throw new HttpException(
110:                     'Responses must be strings or implement Response'
111:                 );
112:             }
113: 
114:             $data[] = array(
115:                 'statusCode' => $response->getStatusCode(),
116:                 'reasonPhrase' => $response->getReasonPhrase(),
117:                 'headers' => $response->getHeaders()->getAll(),
118:                 'body' => $response->getBody(true)
119:             );
120:         }
121: 
122:         $request = $this->client->put('guzzle-server/responses', null, json_encode($data));
123:         $response = $request->send();
124:         
125:         return $response->getStatusCode() == 200;
126:     }
