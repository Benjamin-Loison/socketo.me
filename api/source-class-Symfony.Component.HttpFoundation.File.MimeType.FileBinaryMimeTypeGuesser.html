56:     public function guess($path)
57:     {
58:         if (!is_file($path)) {
59:             throw new FileNotFoundException($path);
60:         }
61: 
62:         if (!is_readable($path)) {
63:             throw new AccessDeniedException($path);
64:         }
65: 
66:         if (!self::isSupported()) {
67:             return null;
68:         }

70:         ob_start();

73:         passthru(sprintf($this->cmd, escapeshellarg($path)), $return);
74:         if ($return > 0) {
75:             ob_end_clean();
76: 
77:             return null;
78:         }
79:
