 89:         if ($file instanceof UploadedFile) {

 91:         }

 94:         if (is_array($file)) {

 96:             sort($keys);

112:     /**

115:      * PHP has a bug that the format of the $_FILES array differs, depending on

117:      * field names ("normal" vs. "parent[child]").
118:      *
119:      * This method fixes the array to look like the "normal" $_FILES array.
120:      *
121:      * It's safe to pass an already converted array, in which case this method
122:      * just returns the original array unmodified.
123:      *
124:      * @param  array $data
125:      *
126:      * @return array
127:      */
128:     protected function fixPhpFilesArray($data)
129:     {
130:         if (!is_array($data)) {
131:             return $data;
132:         }
133: 
134:         $keys = array_keys($data);
135:         sort($keys);
136: 
137:         if (self::$fileKeys != $keys || !isset($data['name']) || !is_array($data['name'])) {
138:             return $data;
139:         }

141:         $files = $data;
142:         foreach (self::$fileKeys as $k) {
143:             unset($files[$k]);
144:         }
145: 
146:         foreach (array_keys($data['name']) as $key) {
147:             $files[$key] = $this->fixPhpFilesArray(array(
148:                 'error'    => $data['error'][$key],
149:                 'name'     => $data['name'][$key],
150:                 'type'     => $data['type'][$key],
151:                 'tmp_name' => $data['tmp_name'][$key],
152:                 'size'     => $data['size'][$key]

154:         }

156:         return $files;
