 193:     /**
 194:      * Prepares the Response before it is sent to the client.
 195:      *
 196:      * This method tweaks the Response to ensure that it is
 197:      * compliant with RFC 2616. Most of the changes are based on
 198:      * the Request that is "associated" with this Response.
 199:      *
 200:      * @param Request $request A Request instance
 201:      *
 202:      * @return Response The current response.
 203:      */
 204:     public function prepare(Request $request)
 205:     {
 206:         $headers = $this->headers;
 207: 
 208:         if ($this->isInformational() || in_array($this->statusCode, array(204, 304))) {
 209:             $this->setContent('');
 210:         }
 211: 
 212:         // Content-type based on the Request
 213:         if (!$headers->has('Content-Type')) {
 214:             $format = $request->getRequestFormat();
 215:             if (null !== $format && $mimeType = $request->getMimeType($format)) {
 216:                 $headers->set('Content-Type', $mimeType);
 217:             }
 218:         }
 219: 
 220:         // Fix Content-Type
 221:         $charset = $this->charset ?: 'UTF-8';
 222:         if (!$headers->has('Content-Type')) {
 223:             $headers->set('Content-Type', 'text/html; charset='.$charset);
 224:         } elseif (0 === strpos($headers->get('Content-Type'), 'text/') && false === strpos($headers->get('Content-Type'), 'charset')) {
 225:             // add the charset
 226:             $headers->set('Content-Type', $headers->get('Content-Type').'; charset='.$charset);
 227:         }
 228: 
 229:         // Fix Content-Length
 230:         if ($headers->has('Transfer-Encoding')) {
 231:             $headers->remove('Content-Length');
 232:         }
 233: 
 234:         if ('HEAD' === $request->getMethod()) {
 235:             // cf. RFC2616 14.13
 236:             $length = $headers->get('Content-Length');
 237:             $this->setContent('');
 238:             if ($length) {
 239:                 $headers->set('Content-Length', $length);
 240:             }
 241:         }
 242: 
 243:         return $this;
 244:     }

 246:     /**
 247:      * Sends HTTP headers.
 248:      *
 249:      * @return Response
 250:      */
 251:     public function sendHeaders()
 252:     {
 253:         // headers have already been sent by the developer
 254:         if (headers_sent()) {
 255:             return $this;
 256:         }
 257: 
 258:         // status
 259:         header(sprintf('HTTP/%s %s %s', $this->version, $this->statusCode, $this->statusText));
 260: 
 261:         // headers
 262:         foreach ($this->headers->all() as $name => $values) {
 263:             foreach ($values as $value) {
 264:                 header($name.': '.$value, false);
 265:             }
 266:         }
 267: 
 268:         // cookies
 269:         foreach ($this->headers->getCookies() as $cookie) {
 270:             setcookie($cookie->getName(), $cookie->getValue(), $cookie->getExpiresTime(), $cookie->getPath(), $cookie->getDomain(), $cookie->isSecure(), $cookie->isHttpOnly());
 271:         }
 272: 
 273:         return $this;
 274:     }
