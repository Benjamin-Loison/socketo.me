 50:     public function fromMessage($message)
 51:     {
 52:         $parsed = ParserRegistry::getInstance()->getParser('message')->parseRequest($message);
 53: 
 54:         if (!$parsed) {
 55:             return false;
 56:         }
 57: 
 58:         $request = $this->fromParts($parsed['method'], $parsed['request_url'],
 59:             $parsed['headers'], $parsed['body'], $parsed['protocol'],
 60:             $parsed['version']);
 61:

 63:         // requests. This factory method should accurately reflect the message, so here we are removing the Expect
 64:         // header if one was not supplied in the message.
 65:         if (!isset($parsed['headers']['Expect']) && !isset($parsed['headers']['expect'])) {
 66:             $request->removeHeader('Expect');

 84:     public function create($method, $url, $headers = null, $body = null, array $options = array())
 85:     {
 86:         $method = strtoupper($method);
 87: 
 88:         if ($method == 'GET' || $method == 'HEAD' || $method == 'TRACE' || $method == 'OPTIONS') {
 89:             // Handle non-entity-enclosing request methods
 90:             $request = new $this->requestClass($method, $url, $headers);
 91:             if ($body) {
 92:                 // The body is where the response body will be stored
 93:                 $type = gettype($body);
 94:                 if ($type == 'string' || $type == 'resource' || $type == 'object') {
 95:                     $request->setResponseBody($body);
 96:                 }
 97:             }
 98:         } else {
 99:             // Create an entity enclosing request by default
100:             $request = new $this->entityEnclosingRequestClass($method, $url, $headers);
101:             if ($body) {
102:                 // Add POST fields and files to an entity enclosing request if an array is used
103:                 if (is_array($body) || $body instanceof Collection) {
104:                     // Normalize PHP style cURL uploads with a leading '@' symbol
105:                     foreach ($body as $key => $value) {
106:                         if (is_string($value) && substr($value, 0, 1) == '@') {
107:                             $request->addPostFile($key, $value);
108:                             unset($body[$key]);
109:                         }
110:                     }
111:                     // Add the fields if they are still present and not all files
112:                     $request->addPostFields($body);
113:                 } else {
114:                     // Add a raw entity body body to the request
115:                     $request->setBody($body, (string) $request->getHeader('Content-Type'));
116:                     if ((string) $request->getHeader('Transfer-Encoding') == 'chunked') {
117:                         $request->removeHeader('Content-Length');
118:                     }
119:                 }
120:             }
121:         }
122: 
123:         if ($options) {
124:             $this->applyOptions($request, $options);
125:         }
126: 
127:         return $request;
128:     }

130:     /**
131:      * Clone a request while changing the method. Emulates the behavior of
132:      * {@see Guzzle\Http\Message\Request::clone}, but can change the HTTP method.
133:      *
134:      * @param RequestInterface $request Request to clone
135:      * @param string           $method  Method to set
136:      *
137:      * @return RequestInterface
138:      */
139:     public function cloneRequestWithMethod(RequestInterface $request, $method)
140:     {
141:         // Create the request with the same client if possible
142:         if ($client = $request->getClient()) {
143:             $cloned = $request->getClient()->createRequest($method, $request->getUrl(), $request->getHeaders());
144:         } else {
145:             $cloned = $this->create($method, $request->getUrl(), $request->getHeaders());
146:         }
147: 
148:         $cloned->getCurlOptions()->replace($request->getCurlOptions()->toArray());
149:         $cloned->setEventDispatcher(clone $request->getEventDispatcher());
150:         // Ensure that that the Content-Length header is not copied if changing to GET or HEAD
151:         if (!($cloned instanceof EntityEnclosingRequestInterface)) {
152:             $cloned->removeHeader('Content-Length');
153:         } elseif ($request instanceof EntityEnclosingRequestInterface) {
154:             $cloned->setBody($request->getBody());
155:         }
156:         $cloned->getParams()->replace($request->getParams()->toArray());
157:         $cloned->dispatch('request.clone', array('request' => $cloned));
158: 
159:         return $cloned;
